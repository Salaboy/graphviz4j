006package org.anarres.graphviz.builder;
007
008import com.google.common.base.Preconditions;
009import java.io.File;
010import java.io.IOException;
011import javax.annotation.Nonnull;
012
013/**
014 * Easy utilities for emitting GraphViz graphs.
015 *
016 * @author shevek
017 */
018public class GraphVizUtils {
019
020    /**
021     * This supports a one-liner: GraphVizUtils.toGraphVizGraph(object).writeTo(new File(...)).
022     *
023     * @param object The object to graph.
024     * @return A newly constructed GraphVizGraph.
025     */
026    @Nonnull
027    public static GraphVizGraph toGraphVizGraph(@Nonnull GraphVizable object) {
028        Preconditions.checkNotNull(object, "GraphVizable object was null.");
029        GraphVizGraph graph = new GraphVizGraph();
030        object.toGraphViz(graph);
031        return graph;
032    }
033
034    /**
035     * Equivalent to {@link #toGraphVizGraph(GraphVizable) toGraphVizGraph}(object).{@link GraphVizGraph#writeTo(java.io.File) writeTo}(file).
036     *
037     * @param file The file to which to write the graph.
038     * @param object The object to convert.
039     * @throws IOException If an I/O error occurs.
040     */
041    public static void toGraphVizFile(@Nonnull File file, @Nonnull GraphVizable object) throws IOException {
042        toGraphVizGraph(object).writeTo(file);
043    }
044}
