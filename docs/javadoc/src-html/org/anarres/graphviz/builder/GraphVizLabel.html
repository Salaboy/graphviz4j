005package org.anarres.graphviz.builder;
006
007import java.io.Serializable;
008import javax.annotation.CheckForNull;
009import javax.annotation.Nonnegative;
010import javax.annotation.Nonnull;
011
012/**
013 *
014 * @author shevek
015 */
016public class GraphVizLabel implements Appendable, Serializable {
017
018    private final StringBuilder buf = new StringBuilder();
019
020    @Nonnegative
021    public int length() {
022        return buf.length();
023    }
024
025    public boolean isEmpty() {
026        return length() == 0;
027    }
028
029    @Nonnull
030    public StringBuilder getBuffer() {
031        return buf;
032    }
033
034    @Nonnull
035    public GraphVizLabel set(@Nonnull CharSequence value) {
036        buf.setLength(0);
037        return append(value);
038    }
039
040    // Optimized case for ToStringBuilder removed as did not filter special characters.
041    @Nonnull
042    public GraphVizLabel append(@CheckForNull Object object) {
043        return append(String.valueOf(object));
044    }
045
046    @Nonnull
047    @Override
048    public GraphVizLabel append(@Nonnull CharSequence csq) {
049        return append(csq, 0, csq.length());
050    }
051
052    @Nonnull
053    @Override
054    public GraphVizLabel append(@Nonnull CharSequence csq, @Nonnegative int start, @Nonnegative int end) {
055        if (end > csq.length())
056            end = csq.length();
057        for (int i = start; i < end; i++) {
058            append(csq.charAt(i));
059        }
060        return this;
061    }
062
063    @Nonnull
064    @Override
065    public GraphVizLabel append(char c) {
066        switch (c) {
067            case '\\':
068            case '"':
069            case '{':
070            case '}':
071                buf.append("\\");
072                buf.append(c);
073                break;
074            case '<':
075                buf.append("&lt;");
076                break;
077            case '>':
078                buf.append("&gt;");
079                break;
080            case '&':
081                buf.append("&amp;");
082                break;
083            case '\n':
084                buf.append("\\l");
085                break;
086            case '\r':
087                break;
088            default:
089                buf.append(c);
090                break;
091        }
092        return this;
093    }
094
095    @Nonnull
096    @Override
097    public String toString() {
098        return buf.toString();
099    }
100}
